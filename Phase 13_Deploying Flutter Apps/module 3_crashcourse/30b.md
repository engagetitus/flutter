













### 30b_App Theme

**Objective:** Learn how to manage and apply themes in a Flutter application to ensure a consistent look and feel.

#### Key Points

- **Themes**: Centralized definition of visual aspects such as colors, text styles, and other UI elements.
- **Light and Dark Themes**: Support for both light and dark modes.
- **Custom Themes**: Define custom themes that align with the appâ€™s brand identity.

#### Implementation

1. **Define a Basic Theme**:
   - Example in `main.dart`:

     ```dart
     import 'package:flutter/material.dart';

     void main() {
       runApp(MyApp());
     }

     class MyApp extends StatelessWidget {
       @override
       Widget build(BuildContext context) {
         return MaterialApp(
           title: 'Theming Example',
           theme: ThemeData(
             primarySwatch: Colors.blue,
             accentColor: Colors.amber,
             textTheme: TextTheme(
               bodyText1: TextStyle(color: Colors.blue, fontSize: 18),
               bodyText2: TextStyle(color: Colors.black, fontSize: 16),
             ),
             buttonTheme: ButtonThemeData(
               buttonColor: Colors.blue,
               textTheme: ButtonTextTheme.primary,
             ),
           ),
           home: HomePage(),
         );
       }
     }

     class HomePage extends StatelessWidget {
       @override
       Widget build(BuildContext context) {
         return Scaffold(
           appBar: AppBar(
             title: Text('Home Page'),
           ),
           body: Center(
             child: Column(
               mainAxisAlignment: MainAxisAlignment.center,
               children: [
                 Text('Hello, World!', style: Theme.of(context).textTheme.bodyText1),
                 SizedBox(height: 20),
                 ElevatedButton(
                   onPressed: () {},
                   child: Text('Press Me'),
                 ),
               ],
             ),
           ),
         );
       }
     }
     ```

2. **Light and Dark Themes**:
   - Define both light and dark themes and switch between them:

     ```dart
     import 'package:flutter/material.dart';

     void main() {
       runApp(MyApp());
     }

     class MyApp extends StatelessWidget {
       @override
       Widget build(BuildContext context) {
         return MaterialApp(
           title: 'Theming Example',
           theme: ThemeData.light().copyWith(
             primaryColor: Colors.blue,
             accentColor: Colors.amber,
             textTheme: TextTheme(
               bodyText1: TextStyle(color: Colors.blue, fontSize: 18),
               bodyText2: TextStyle(color: Colors.black, fontSize: 16),
             ),
             buttonTheme: ButtonThemeData(
               buttonColor: Colors.blue,
               textTheme: ButtonTextTheme.primary,
             ),
           ),
           darkTheme: ThemeData.dark().copyWith(
             primaryColor: Colors.blue,
             accentColor: Colors.amber,
             textTheme: TextTheme(
               bodyText1: TextStyle(color: Colors.white, fontSize: 18),
               bodyText2: TextStyle(color: Colors.grey, fontSize: 16),
             ),
             buttonTheme: ButtonThemeData(
               buttonColor: Colors.blue,
               textTheme: ButtonTextTheme.primary,
             ),
           ),
           themeMode: ThemeMode.system,
           home: HomePage(),
         );
       }
     }

     class HomePage extends StatelessWidget {
       @override
       Widget build(BuildContext context) {
         return Scaffold(
           appBar: AppBar(
             title: Text('Home Page'),
           ),
           body: Center(
             child: Column(
               mainAxisAlignment: MainAxisAlignment.center,
               children: [
                 Text('Hello, World!', style: Theme.of(context).textTheme.bodyText1),
                 SizedBox(height: 20),
                 ElevatedButton(
                   onPressed: () {},
                   child: Text('Press Me'),
                 ),
               ],
             ),
           ),
         );
       }
     }
     ```

3. **Custom Themes**:
   - Define a custom theme for your application:

     ```dart
     import 'package:flutter/material.dart';

     void main() {
       runApp(MyApp());
     }

     class MyApp extends StatelessWidget {
       @override
       Widget build(BuildContext context) {
         return MaterialApp(
           title: 'Theming Example',
           theme: ThemeData(
             primaryColor: Colors.teal,
             accentColor: Colors.orange,
             scaffoldBackgroundColor: Colors.teal.shade50,
             appBarTheme: AppBarTheme(
               color: Colors.teal.shade700,
               textTheme: TextTheme(
                 headline6: TextStyle(color: Colors.white, fontSize: 20),
               ),
             ),
             textTheme: TextTheme(
               bodyText1: TextStyle(color: Colors.teal.shade900, fontSize: 18),
               bodyText2: TextStyle(color: Colors.teal.shade600, fontSize: 16),
             ),
             buttonTheme: ButtonThemeData(
               buttonColor: Colors.teal,
               textTheme: ButtonTextTheme.primary,
             ),
             cardTheme: CardTheme(
               color: Colors.teal.shade100,
               elevation: 5,
               shape: RoundedRectangleBorder(
                 borderRadius: BorderRadius.circular(10),
               ),
             ),
           ),
           home: HomePage(),
         );
       }
     }

     class HomePage extends StatelessWidget {
       @override
       Widget build(BuildContext context) {
         return Scaffold(
           appBar: AppBar(
             title: Text('Home Page'),
           ),
           body: Center(
             child: Column(
               mainAxisAlignment: MainAxisAlignment.center,
               children: [
                 Text('Welcome to the Custom Themed App!', style: Theme.of(context).textTheme.bodyText1),
                 SizedBox(height: 20),
                 ElevatedButton(
                   onPressed: () {},
                   child: Text('Press Me'),
                 ),
                 SizedBox(height: 20),
                 Card(
                   child: Padding(
                     padding: const EdgeInsets.all(16.0),
                     child: Text('This is a card', style: Theme.of(context).textTheme.bodyText2),
                   ),
                 ),
               ],
             ),
           ),
         );
       }
     }
     ```

4. **Dynamic Theme Switching**:
   - Add functionality to switch themes dynamically within the app:

     ```dart
     import 'package:flutter/material.dart';

     void main() {
       runApp(MyApp());
     }

     class MyApp extends StatefulWidget {
       @override
       _MyAppState createState() => _MyAppState();
     }

     class _MyAppState extends State<MyApp> {
       ThemeMode _themeMode = ThemeMode.light;

       void _toggleTheme() {
         setState(() {
           _themeMode = _themeMode == ThemeMode.light ? ThemeMode.dark : ThemeMode.light;
         });
       }

       @override
       Widget build(BuildContext context) {
         return MaterialApp(
           title: 'Theming Example',
           theme: ThemeData.light().copyWith(
             primaryColor: Colors.blue,
             accentColor: Colors.amber,
             textTheme: TextTheme(
               bodyText1: TextStyle(color: Colors.blue, fontSize: 18),
               bodyText2: TextStyle(color: Colors.black, fontSize: 16),
             ),
             buttonTheme: ButtonThemeData(
               buttonColor: Colors.blue,
               textTheme: ButtonTextTheme.primary,
             ),
           ),
           darkTheme: ThemeData.dark().copyWith(
             primaryColor: Colors.blue,
             accentColor: Colors.amber,
             textTheme: TextTheme(
               bodyText1: TextStyle(color: Colors.white, fontSize: 18),
               bodyText2: TextStyle(color: Colors.grey, fontSize: 16),
             ),
             buttonTheme: ButtonThemeData(
               buttonColor: Colors.blue,
               textTheme: ButtonTextTheme.primary,
             ),
           ),
           themeMode: _themeMode,
           home: HomePage(toggleTheme: _toggleTheme),
         );
       }
     }

     class HomePage extends StatelessWidget {
       final VoidCallback toggleTheme;

       HomePage({required this.toggleTheme});

       @override
       Widget build(BuildContext context) {
         return Scaffold(
           appBar: AppBar(
             title: Text('Home Page'),
             actions: [
               IconButton(
                 icon: Icon(Icons.brightness_6),
                 onPressed: toggleTheme,
               ),
             ],
           ),
           body: Center(
             child: Column(
               mainAxisAlignment: MainAxisAlignment.center,
               children: [
                 Text('Hello, World!', style: Theme.of(context).textTheme.bodyText1),
                 SizedBox(height: 20),
                 ElevatedButton(
                   onPressed: () {},
                   child: Text('Press Me'),
                 ),
               ],
             ),
           ),
         );
       }
     }
     ```
